<!DOCTYPE html>
        <html>
        <head>
            <title>Parzystość liczby</title>
        </head>
        <body>
        
        <script>
            var liczba = prompt("Podaj liczbę:");
            liczba = parseInt(liczba);
            if (liczba % 2 === 0) {
                document.write("<font color=red size=3>liczba jest parzysta</font><br>");
                document.write("<p style="font-size: 24px; color: red;">liczba jest parzysta</p>");
            } else {
                document.write("<font color=silver size=5>liczba jest nieparzysta</font><br>");
                document.write("<p style="font-size: 40px; color: silver;">liczba jest nieparzysta</p>");
            }
        </script>
        
        </body>
        </html>
