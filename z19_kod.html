                function Pole_Graczyk() {
                    var a_gra = parseFloat(document.getElementById('dlugosc').value);
                    var b_gra = a_gra * a_gra;
                    document.getElementById("nazwa").innerHTML = "Pole kwadratu= " + b_gra.toFixed(2) + " cm^2";
                }
                function obwod_Graczyk() {
                    var a_gra = parseFloat(document.getElementById('dlugosc').value);
                    var b_gra = 4 * a_gra;
                    document.getElementById("nazwa").innerHTML = "Obwód kwadratu= " + b_gra.toFixed(2) + " cm";
                }
            </script>
