            <script>
                var autobus = { 
                    marka: 'Mercedes', 
                    model: 'Citaro', 
                    ilu_pasazerow: 80, 
                    numer_linii: '105',
                    czy_miejski: true,
                    wyswietl_autobus: function () { 
                        document.write(this.marka + ' ' + this.model); 
                    } 
                }; 
                autobus.wyswietl_autobus(); 
                
                autobus.rok_produkcji = 2015;
                autobus.wypisz = function() {
                    alert(this.marka + " został wyprodukowany w " + this.rok_produkcji + " roku");
                };
                autobus.wypisz();
            </script>
